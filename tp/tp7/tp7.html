Code HTML donné :

<!DOCTYPE html>
<html>
  <head>

      .logo { display:inline-block; border:1px solid chocolate; margin:1px; }
      .emploi { display:inline-block; border:1px solid chartreuse; }

      .pdp { margin-top:2px; margin-bottom:2px; background-color:chartreuse; height:50px; }
    </style>
  </head>
  <body>
    <header class="entete">
      <div>
        <nav class="menu1">MENU 1</nav>

    </header>

      <article>ARTICLE</article>
    </section>
    <footer class="pdp">PIED DE PAGE</footer>
  </body>
</html>
